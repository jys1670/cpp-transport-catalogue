91SphereProjector::SphereProjector(PointInputIt points_begin,
92 PointInputIt points_end, double max_width,
93 double max_height, double padding)
94 : padding_(padding) {
95
96 if (points_begin == points_end) {
97 return;
98 }
99
100 const auto [left_it, right_it] =
101 std::minmax_element(points_begin, points_end,
102 [](auto lhs, auto rhs) { return lhs.lng < rhs.lng; });
103 min_lon_ = left_it->lng;
104 const double max_lon = right_it->lng;
105
106 const auto [bottom_it, top_it] =
107 std::minmax_element(points_begin, points_end,
108 [](auto lhs, auto rhs) { return lhs.lat < rhs.lat; });
109 const double min_lat = bottom_it->lat;
110 max_lat_ = top_it->lat;
111
112 std::optional<double> width_zoom;
113 if (!IsZero(max_lon - min_lon_)) {
114 width_zoom = (max_width - 2 * padding) / (max_lon - min_lon_);
115 }
116
117 std::optional<double> height_zoom;
118 if (!IsZero(max_lat_ - min_lat)) {
119 height_zoom = (max_height - 2 * padding) / (max_lat_ - min_lat);
120 }
121
122 if (width_zoom && height_zoom) {
123 zoom_coeff_ = std::min(*width_zoom, *height_zoom);
124 } else if (width_zoom) {
125 zoom_coeff_ = *width_zoom;
126 } else if (height_zoom) {
127 zoom_coeff_ = *height_zoom;
128 }
129}
