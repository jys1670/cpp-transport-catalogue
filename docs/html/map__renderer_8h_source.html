126SphereProjector::SphereProjector(PointInputIt points_begin,
127 PointInputIt points_end, double max_width,
128 double max_height, double padding)
129 : padding_(padding) {
130
131 if (points_begin == points_end) {
132 return;
133 }
134
135 const auto [left_it, right_it] =
136 std::minmax_element(points_begin, points_end,
137 [](auto lhs, auto rhs) { return lhs.lng < rhs.lng; });
138 min_lon_ = left_it->lng;
139 const double max_lon = right_it->lng;
140
141 const auto [bottom_it, top_it] =
142 std::minmax_element(points_begin, points_end,
143 [](auto lhs, auto rhs) { return lhs.lat < rhs.lat; });
144 const double min_lat = bottom_it->lat;
145 max_lat_ = top_it->lat;
146
147 std::optional<double> width_zoom;
148 if (!IsZero(max_lon - min_lon_)) {
149 width_zoom = (max_width - 2 * padding) / (max_lon - min_lon_);
150 }
151
152 std::optional<double> height_zoom;
153 if (!IsZero(max_lat_ - min_lat)) {
154 height_zoom = (max_height - 2 * padding) / (max_lat_ - min_lat);
155 }
156
157 if (width_zoom && height_zoom) {
158 zoom_coeff_ = std::min(*width_zoom, *height_zoom);
159 } else if (width_zoom) {
160 zoom_coeff_ = *width_zoom;
161 } else if (height_zoom) {
162 zoom_coeff_ = *height_zoom;
163 }
164}
