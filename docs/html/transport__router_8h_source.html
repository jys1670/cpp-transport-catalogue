114void TransportRouter::InsertEdgesBetweenStops(InputIt begin, InputIt end,

116 for (auto it1 = begin; it1 != end; ++it1) {
117 double tot_dist{0};
118 for (auto it2 = next(it1); it2 != end; ++it2) {
119 auto norm_id = vertex_to_id_.at(Vertex().SetStop(*it1).SetWait(false));
120 auto wait_id = vertex_to_id_.at(Vertex().SetStop(*it2).SetWait(true));
121 size_t stops_between = std::distance(it1, it2);
122 auto curr_dist =
123 catalogue_.GetStopsRealDist((*prev(it2))->name, (*it2)->name).value();
124 tot_dist += curr_dist;
125 graph_.AddEdge(
126 {norm_id, wait_id, tot_dist / bus_velocity_, bus, stops_between});
