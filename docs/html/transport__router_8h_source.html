92void TransportRouter::InsertEdgesBetweenStops(InputIt begin, InputIt end,
93 const data::Bus *bus) {
94 for (auto it1 = begin; it1 != end; ++it1) {
95 double tot_dist{0};
96 for (auto it2 = next(it1); it2 != end; ++it2) {
97 auto norm_id = vertex_to_id_.at(Vertex().SetStop(*it1).SetWait(false));
98 auto wait_id = vertex_to_id_.at(Vertex().SetStop(*it2).SetWait(true));
99 auto stops_between = static_cast<size_t>(std::distance(it1, it2));
100 auto curr_dist =
101 catalogue_.GetStopsRealDist((*prev(it2))->name, (*it2)->name).value();
102 tot_dist += curr_dist;
103 graph_.AddEdge(Edge()
104 .SetFromVertex(norm_id)
105 .SetToVertex(wait_id)
106 .SetWeight(tot_dist / settings_.bus_velocity)
107 .SetBus(bus)
108 .SetStopCount(stops_between));
109 }
110 }
